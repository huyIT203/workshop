Database Configuration

MongoDB Setup

1. Create Database and User

# Connect to MongoDB
mongosh

# Create database
use elearning

# Create user with appropriate permissions
db.createUser({
  user: "elearning_user",
  pwd: "secure_password_123",
  roles: [
    { role: "readWrite", db: "elearning" },
    { role: "dbAdmin", db: "elearning" }
  ]
})

# Verify user creation
db.getUsers()

2. MongoDB Configuration in application.properties

# MongoDB Configuration
spring.data.mongodb.uri=mongodb://elearning_user:secure_password_123@localhost:27017/elearning
spring.data.mongodb.database=elearning

# MongoDB Connection Pool
spring.data.mongodb.max-connection-pool-size=100
spring.data.mongodb.min-connection-pool-size=5
spring.data.mongodb.max-connection-idle-time=30000
MySQL Setup

1. Create Database and User

-- Connect to MySQL as root
mysql -u root -p

-- Create database
CREATE DATABASE elearning CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- Create user
CREATE USER 'elearning_user'@'localhost' IDENTIFIED BY 'secure_password_123';

-- Grant permissions
GRANT ALL PRIVILEGES ON elearning.* TO 'elearning_user'@'localhost';
GRANT SELECT, INSERT, UPDATE, DELETE ON elearning.* TO 'elearning_user'@'localhost';

-- Apply changes
FLUSH PRIVILEGES;

-- Verify
SHOW DATABASES;
SELECT User, Host FROM mysql.user WHERE User = 'elearning_user';

2. MySQL Configuration in application.properties

# MySQL Configuration
spring.datasource.url=jdbc:mysql://localhost:3306/elearning?useSSL=false&serverTimezone=UTC&allowPublicKeyRetrieval=true
spring.datasource.username=elearning_user
spring.datasource.password=secure_password_123
spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver

# JPA/Hibernate Configuration
spring.jpa.hibernate.ddl-auto=validate
spring.jpa.show-sql=true
spring.jpa.properties.hibernate.dialect=org.hibernate.dialect.MySQL8Dialect
spring.jpa.properties.hibernate.format_sql=true

# Connection Pool
spring.datasource.hikari.maximum-pool-size=20
spring.datasource.hikari.minimum-idle=5
spring.datasource.hikari.connection-timeout=30000
