Backup and Recovery Configuration

MongoDB Backup

Automated Backup Script

#!/bin/bash
# /usr/local/bin/mongodb-backup.sh

BACKUP_DIR="/backup/mongodb"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="elearning"

# Create backup directory
mkdir -p $BACKUP_DIR

# Perform backup
mongodump --db $DB_NAME --out $BACKUP_DIR/$DATE

# Compress backup
tar -czf $BACKUP_DIR/$DATE.tar.gz -C $BACKUP_DIR $DATE

# Remove uncompressed backup
rm -rf $BACKUP_DIR/$DATE

# Keep only last 7 days of backups
find $BACKUP_DIR -name "*.tar.gz" -mtime +7 -delete

echo "MongoDB backup completed: $BACKUP_DIR/$DATE.tar.gz"

Backup Cron Job

# Add to crontab
crontab -e

# Daily backup at 2 AM
0 2 * * * /usr/local/bin/mongodb-backup.sh >> /var/log/mongodb-backup.log 2>&1
MySQL Backup

Automated Backup Script

#!/bin/bash
# /usr/local/bin/mysql-backup.sh

BACKUP_DIR="/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="elearning"
DB_USER="elearning_user"
DB_PASS="secure_password_123"

# Create backup directory
mkdir -p $BACKUP_DIR

# Perform backup
mysqldump -u $DB_USER -p$DB_PASS --single-transaction --routines --triggers $DB_NAME > $BACKUP_DIR/$DATE.sql

# Compress backup
gzip $BACKUP_DIR/$DATE.sql

# Keep only last 7 days of backups
find $BACKUP_DIR -name "*.sql.gz" -mtime +7 -delete

echo "MySQL backup completed: $BACKUP_DIR/$DATE.sql.gz"

Monitoring and Health Checks

MongoDB Monitoring

Health Check Script

#!/bin/bash
# /usr/local/bin/mongodb-health-check.sh

# Check MongoDB service status
if systemctl is-active --quiet mongod; then
    echo "MongoDB service is running"
    
    # Check database connectivity
    if mongosh --eval "db.runCommand('ping')" > /dev/null 2>&1; then
        echo "MongoDB is responding to queries"
        
        # Check database size
        DB_SIZE=$(mongosh --quiet --eval "db.stats().dataSize")
        echo "Database size: $DB_SIZE bytes"
        
        # Check connection count
        CONN_COUNT=$(mongosh --quiet --eval "db.serverStatus().connections.current")
        echo "Current connections: $CONN_COUNT"
    else
        echo "ERROR: MongoDB is not responding to queries"
        exit 1
    fi
else
    echo "ERROR: MongoDB service is not running"
    exit 1
fi
