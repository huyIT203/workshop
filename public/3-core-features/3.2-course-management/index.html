@Service
@Transactional
public class CoursePublicationService {
    
    public PublicationRequest submitForReview(String courseId, String teacherId) {
        Course course = courseService.findById(courseId);
        
        // Validate course completeness
        validateCourseCompleteness(course);
        
        // Create publication request
        PublicationRequest request = new PublicationRequest();
        request.setCourseId(courseId);
        request.setTeacherId(teacherId);
        request.setStatus(PublicationStatus.PENDING_REVIEW);
        request.setSubmittedAt(LocalDateTime.now());
        
        // Notify administrators
        notificationService.notifyAdminsOfPublicationRequest(request);
        
        return publicationRequestRepository.save(request);
    }
    
    public void approveCourse(String courseId, String adminId, String comments) {
        Course course = courseService.findById(courseId);
        PublicationRequest request = publicationRequestRepository.findByCourseId(courseId);
        
        // Update course status
        course.setStatus(CourseStatus.PUBLISHED);
        course.setPublishedAt(LocalDateTime.now());
        course.setApprovedBy(adminId);
        
        // Update publication request
        request.setStatus(PublicationStatus.APPROVED);
        request.setReviewedBy(adminId);
        request.setReviewedAt(LocalDateTime.now());
        request.setComments(comments);
        
        // Notify teacher
        notificationService.notifyTeacherOfCourseApproval(course.getTeacher().getEmail());
        
        // Save changes
        courseService.save(course);
        publicationRequestRepository.save(request);
    }
    
    private void validateCourseCompleteness(Course course) {
        if (course.getModules().isEmpty()) {
            throw new CourseValidationException("Course must have at least one module");
        }
        
        for (Module module : course.getModules()) {
            if (module.getLessons().isEmpty()) {
                throw new CourseValidationException("Module '" + module.getTitle() + "' must have at least one lesson");
            }
        }
        
        if (course.getEstimatedHours() == null || course.getEstimatedHours() <= 0) {
            throw new CourseValidationException("Course must have estimated duration");
        }
        
        if (StringUtils.isEmpty(course.getDescription())) {
            throw new CourseValidationException("Course must have a description");
        }
    }
}

@Service
@Transactional
public class EnrollmentService {
    
    public Enrollment enrollStudent(String courseId, String studentId) {
        Course course = courseService.findById(courseId);
        User student = userService.findById(studentId);
        
        // Check if course is available for enrollment
        validateEnrollmentEligibility(course, student);
        
        // Check if student is already enrolled
        if (enrollmentRepository.existsByCourseIdAndStudentId(courseId, studentId)) {
            throw new AlreadyEnrolledException("Student is already enrolled in this course");
        }
        
        // Create enrollment
        Enrollment enrollment = new Enrollment();
        enrollment.setCourse(course);
        enrollment.setStudent(student);
        enrollment.setEnrollmentDate(LocalDateTime.now());
        enrollment.setStatus(EnrollmentStatus.ACTIVE);
        enrollment.setProgress(0.0);
        
        // Initialize progress tracking
        initializeStudentProgress(course, student);
        
        // Notify teacher of new enrollment
        notificationService.notifyTeacherOfNewEnrollment(course.getTeacher(), student);
        
        // Send welcome email to student
        emailService.sendCourseWelcomeEmail(student.getEmail(), course);
        
        return enrollmentRepository.save(enrollment);
    }
    
    private void validateEnrollmentEligibility(Course course, User student) {
        if (course.getStatus() != CourseStatus.PUBLISHED && course.getStatus() != CourseStatus.ACTIVE) {
            throw new CourseNotAvailableException("Course is not available for enrollment");
        }
        
        if (course.getMaxStudents() != null && 
            enrollmentRepository.countByCourseId(course.getId()) >= course.getMaxStudents()) {
            throw new CourseFullException("Course has reached maximum enrollment capacity");
        }
        
        // Check prerequisites
        if (course.getPrerequisites() != null && !course.getPrerequisites().isEmpty()) {
            validatePrerequisites(course.getPrerequisites(), student);
        }
    }
    
    private void initializeStudentProgress(Course course, User student) {
        for (Module module : course.getModules()) {
            for (Lesson lesson : module.getLessons()) {
                Progress progress = new Progress();
                progress.setStudent(student);
                progress.setLesson(lesson);
                progress.setCompleted(false);
                progress.setScore(0.0);
                progress.setStartedAt(LocalDateTime.now());
                
                progressRepository.save(progress);
            }
        }
    }
}

@Service
@Transactional
public class ProgressTrackingService {
    
    public Progress updateLessonProgress(String lessonId, String studentId, boolean completed, Double score) {
        Progress progress = progressRepository.findByLessonIdAndStudentId(lessonId, studentId)
            .orElseThrow(() -> new ProgressNotFoundException("Progress record not found"));
        
        progress.setCompleted(completed);
        if (score != null) {
            progress.setScore(score);
        }
        
        if (completed) {
            progress.setCompletedAt(LocalDateTime.now());
        }
        
        // Update course-level progress
        updateCourseProgress(studentId, progress.getLesson().getModule().getCourse().getId());
        
        return progressRepository.save(progress);
    }
    
    public CourseProgress getCourseProgress(String courseId, String studentId) {
        Course course = courseService.findById(courseId);
        List<Progress> lessonProgress = progressRepository.findByCourseIdAndStudentId(courseId, studentId);
        
        int totalLessons = course.getModules().stream()
            .mapToInt(module -> module.getLessons().size())
            .sum();
        
        int completedLessons = (int) lessonProgress.stream()
            .filter(Progress::isCompleted)
            .count();
        
        double overallProgress = totalLessons > 0 ? (double) completedLessons / totalLessons * 100 : 0.0;
        
        double averageScore = lessonProgress.stream()
            .mapToDouble(Progress::getScore)
            .average()
            .orElse(0.0);
        
        return CourseProgress.builder()
            .courseId(courseId)
            .studentId(studentId)
            .totalLessons(totalLessons)
            .completedLessons(completedLessons)
            .overallProgress(overallProgress)
            .averageScore(averageScore)
            .lastActivity(LocalDateTime.now())
            .build();
    }
    
    private void updateCourseProgress(String studentId, String courseId) {
        CourseProgress courseProgress = getCourseProgress(courseId, studentId);
        
        // Check if course is completed
        if (courseProgress.getOverallProgress() >= 100.0) {
            Enrollment enrollment = enrollmentRepository.findByCourseIdAndStudentId(courseId, studentId);
            enrollment.setStatus(EnrollmentStatus.COMPLETED);
            enrollment.setCompletedAt(LocalDateTime.now());
            enrollmentRepository.save(enrollment);
            
            // Generate completion certificate
            certificateService.generateCertificate(enrollment);
        }
    }
}

@Service

    public CourseAnalytics getCourseAnalytics(String courseId) {
        Course course = courseService.findById(courseId);

        double completionRate = totalEnrollments > 0 ?

        // Calculate average progress
        double averageProgress = enrollments.stream()
            .mapToDouble(Enrollment::getProgress)
            .average()

        // Calculate average score
        List<Progress> allProgress = progressRepository.findByCourseId(courseId);

            .activeEnrollments(activeEnrollments)
            .completedEnrollments(completedEnrollments)

@Service
public class AdaptiveLearningService {
    
    public LearningPath generatePersonalizedPath(String courseId, String studentId) {
        Course course = courseService.findById(courseId);
        User student = userService.findById(studentId);
        List<Progress> studentProgress = progressRepository.findByCourseIdAndStudentId(courseId, studentId);
        
        // Analyze student performance
        StudentPerformanceAnalysis analysis = analyzeStudentPerformance(studentProgress);
        
        // Generate personalized learning path
        LearningPath learningPath = new LearningPath();
        learningPath.setCourseId(courseId);
        learningPath.setStudentId(studentId);
        learningPath.setGeneratedAt(LocalDateTime.now());
        
        List<LearningStep> steps = new ArrayList<>();
        
        for (Module module : course.getModules()) {
            for (Lesson lesson : module.getLessons()) {
                Progress progress = studentProgress.stream()
                    .filter(p -> p.getLesson().getId().equals(lesson.getId()))
                    .findFirst()
                    .orElse(null);
                
                LearningStep step = createLearningStep(lesson, progress, analysis);
                steps.add(step);
            }
        }
        
        // Reorder steps based on student performance and difficulty
        steps = optimizeLearningSequence(steps, analysis);
        
        learningPath.setSteps(steps);
        return learningPath;
    }
    
    private StudentPerformanceAnalysis analyzeStudentPerformance(List<Progress> progress) {
        double averageScore = progress.stream()
            .mapToDouble(Progress::getScore)
            .average()
            .orElse(0.0);
        
        long completedLessons = progress.stream()
            .filter(Progress::isCompleted)
            .count();
        
        // Determine learning pace
        LearningPace pace = determineLearningPace(progress);
        
        // Identify strengths and weaknesses
        List<String> strengths = identifyStrengths(progress);
        List<String> weaknesses = identifyWeaknesses(progress);
        
        return StudentPerformanceAnalysis.builder()
            .averageScore(averageScore)
            .completedLessons(completedLessons)
            .learningPace(pace)
            .strengths(strengths)
            .weaknesses(weaknesses)
            .build();
    }
    
    private LearningStep createLearningStep(Lesson lesson, Progress progress, StudentPerformanceAnalysis analysis) {
        LearningStep step = new LearningStep();
        step.setLessonId(lesson.getId());
        step.setLessonTitle(lesson.getTitle());
        step.setEstimatedDuration(lesson.getEstimatedDuration());
        step.setDifficulty(lesson.getDifficulty());
        
        if (progress != null) {
            step.setCompleted(progress.isCompleted());
            step.setScore(progress.getScore());
            step.setRecommendedReview(progress.getScore() < 70.0);
        }
        
        // Adjust difficulty based on student performance
        if (analysis.getAverageScore() < 60.0) {
            step.setDifficulty(step.getDifficulty().decrease());
        } else if (analysis.getAverageScore() > 90.0) {
            step.setDifficulty(step.getDifficulty().increase());
        }
        
        return step;
    }
}

@Service
public class ContentDeliveryService {
    
    public ResponseEntity<Resource> streamVideo(String contentId, String range) {
        Content content = contentService.findById(contentId);
        
        if (!content.getContentType().startsWith("video/")) {
            throw new InvalidContentTypeException("Content is not a video");
        }
        
        Resource videoResource = fileStorageService.loadFileAsResource(content.getFilePath());
        
        if (!videoResource.exists()) {
            throw new ContentNotFoundException("Video file not found");
        }
        
        // Handle range requests for video streaming
        if (range != null) {
            return handleRangeRequest(videoResource, range, content);
        }
        
        // Return full video
        return ResponseEntity.ok()
            .contentType(MediaTypeFactory.getMediaType(videoResource).orElse(MediaType.APPLICATION_OCTET_STREAM))
            .header(HttpHeaders.CONTENT_DISPOSITION, "inline; filename=\"" + content.getOriginalFilename() + "\"")
            .body(videoResource);
    }
    
    private ResponseEntity<Resource> handleRangeRequest(Resource videoResource, String range, Content content) {
        try {
            long fileLength = videoResource.contentLength();
            long start = 0, end = fileLength - 1;
            
            if (range.startsWith("bytes=")) {
                String[] ranges = range.substring(6).split("-");
                start = Long.parseLong(ranges[0]);
                if (ranges.length > 1 && !ranges[1].isEmpty()) {
                    end = Long.parseLong(ranges[1]);
                }
            }
            
            long contentLength = end - start + 1;
            
            // Create partial content response
            return ResponseEntity.status(HttpStatus.PARTIAL_CONTENT)
                .contentType(MediaTypeFactory.getMediaType(videoResource).orElse(MediaType.APPLICATION_OCTET_STREAM))
                .header(HttpHeaders.ACCEPT_RANGES, "bytes")
                .header(HttpHeaders.CONTENT_RANGE, "bytes " + start + "-" + end + "/" + fileLength)
                .header(HttpHeaders.CONTENT_LENGTH, String.valueOf(contentLength))
                .body(new InputStreamResource(videoResource.getInputStream()));
                
        } catch (IOException e) {
            throw new ContentDeliveryException("Error processing range request", e);
        }
    }
}

@Service
public class CourseSearchService {
    
    public Page<Course> searchCourses(CourseSearchCriteria criteria, Pageable pageable) {
        Specification<Course> spec = Specification.where(null);
        
        // Title and description search
        if (StringUtils.hasText(criteria.getSearchTerm())) {
            spec = spec.and(CourseSpecifications.hasSearchTerm(criteria.getSearchTerm()));
        }
        
        // Category filter
        if (criteria.getCategory() != null) {
            spec = spec.and(CourseSpecifications.hasCategory(criteria.getCategory()));
        }
        
        // Difficulty level filter
        if (criteria.getDifficulty() != null) {
            spec = spec.and(CourseSpecifications.hasDifficulty(criteria.getDifficulty()));
        }
        
        // Price range filter
        if (criteria.getMinPrice() != null || criteria.getMaxPrice() != null) {
            spec = spec.and(CourseSpecifications.hasPriceRange(criteria.getMinPrice(), criteria.getMaxPrice()));
        }
        
        // Duration filter
        if (criteria.getMaxDuration() != null) {
            spec = spec.and(CourseSpecifications.hasMaxDuration(criteria.getMaxDuration()));
        }
        
        // Rating filter
        if (criteria.getMinRating() != null) {
            spec = spec.and(CourseSpecifications.hasMinRating(criteria.getMinRating()));
        }
        
        // Language filter
        if (criteria.getLanguage() != null) {
            spec = spec.and(CourseSpecifications.hasLanguage(criteria.getLanguage()));
        }
        
        // Status filter (only published courses for students)
        spec = spec.and(CourseSpecifications.isPublished());
        
        return courseRepository.findAll(spec, pageable);
    }
    
    public List<Course> getRecommendedCourses(String studentId, int limit) {
        User student = userService.findById(studentId);
        List<Enrollment> studentEnrollments = enrollmentRepository.findByStudentId(studentId);
        
        // Get student preferences and history
        StudentPreferences preferences = studentPreferenceService.getPreferences(studentId);
        List<String> completedCategories = getCompletedCategories(studentEnrollments);
        
        // Find courses in preferred categories
        List<Course> recommendedCourses = courseRepository.findByCategoryInAndStatus(
            preferences.getPreferredCategories(), CourseStatus.PUBLISHED);
        
        // Filter out already enrolled courses
        recommendedCourses = recommendedCourses.stream()
            .filter(course -> !isEnrolledInCourse(studentEnrollments, course.getId()))
            .collect(Collectors.toList());
        
        // Score and rank courses
        List<CourseRecommendation> scoredCourses = recommendedCourses.stream()
            .map(course -> scoreCourse(course, preferences, completedCategories))
            .sorted(Comparator.comparing(CourseRecommendation::getScore).reversed())
            .limit(limit)
            .map(CourseRecommendation::getCourse)
            .collect(Collectors.toList());
        
        return scoredCourses;
    }
    
    private double scoreCourse(Course course, StudentPreferences preferences, List<String> completedCategories) {
        double score = 0.0;
        
        // Category preference bonus
        if (preferences.getPreferredCategories().contains(course.getCategory())) {
            score += 10.0;
        }
        
        // Difficulty match bonus
        if (preferences.getPreferredDifficulty() == course.getDifficulty()) {
            score += 5.0;
        }
        
        // New category exploration bonus
        if (!completedCategories.contains(course.getCategory())) {
            score += 3.0;
        }
        
        // Rating bonus
        if (course.getAverageRating() != null) {
            score += course.getAverageRating() * 2.0;
        }
        
        // Popularity bonus
        long enrollmentCount = enrollmentRepository.countByCourseId(course.getId());
        score += Math.min(enrollmentCount / 100.0, 5.0);
        
        return score;
    }
