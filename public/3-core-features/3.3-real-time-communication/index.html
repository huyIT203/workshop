Real-time Communication Features

Overview

The Esoclusty E-Learning Platform provides comprehensive real-time communication capabilities that enhance the learning experience through instant messaging, live notifications, and collaborative features. This section explores the WebSocket implementation, chat system, and real-time collaboration tools.

WebSocket Implementation

STOMP Protocol Configuration

WebSocket Configuration

@Configuration
@EnableWebSocketMessageBroker
public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {
    
    @Override
    public void configureMessageBroker(MessageBrokerRegistry config) {
        config.enableSimpleBroker("/topic", "/queue", "/user");
        config.setApplicationDestinationPrefixes("/app");
        config.setUserDestinationPrefix("/user");
    }
    
    @Override
    public void registerStompEndpoints(StompEndpointRegistry registry) {
        registry.addEndpoint("/ws")
            .setAllowedOrigins("*")
            .withSockJS();
    }
    
    @Override
    public boolean configureMessageConverters(List<MessageConverter> messageConverters) {
        messageConverters.add(new MappingJackson2MessageConverter());
        return false;
    }
}

Security Configuration

@Configuration
public class WebSocketSecurityConfig {
    
    @Bean
    public ChannelInterceptor channelInterceptor() {
        return new ChannelInterceptor() {
            @Override
            public Message<?> preSend(Message<?> message, MessageChannel channel) {
                StompHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);
                
                if (StompCommand.CONNECT.equals(accessor.getCommand())) {
                    String token = accessor.getFirstNativeHeader("Authorization");
                    if (token != null && token.startsWith("Bearer ")) {
                        token = token.substring(7);
                        // Validate JWT token and set user
                        String username = jwtService.extractUsername(token);
                        accessor.setUser(new Principal() {
                            @Override
                            public String getName() { return username; }
                        });
                    }
                }
                return message;
            }
        };
    }
}

Live Chat System

Chat Room Management

Chat Room Structure

@Entity
@Table(name = "chat_rooms")
public class ChatRoom {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(nullable = false)
    private String name;
    
    @Enumerated(EnumType.STRING)
    private ChatRoomType type; // COURSE, PRIVATE, GROUP
    
    private String courseId; // For course-specific chats
    private String createdBy;
    private LocalDateTime createdAt;
    
    @OneToMany(mappedBy = "chatRoom", cascade = CascadeType.ALL)
    private List<ChatMessage> messages;
    
    @OneToMany(mappedBy = "chatRoom", cascade = CascadeType.ALL)
    private List<ChatParticipant> participants;
}

@Entity
@Table(name = "chat_messages")
public class ChatMessage {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(nullable = false)
    private String content;
    
    private String senderId;
    private String senderName;
    private LocalDateTime timestamp;
    private MessageType type; // TEXT, FILE, SYSTEM
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "chat_room_id")
    private ChatRoom chatRoom;
    
    private String fileUrl; // For file messages
    private String fileName;
    private Long fileSize;
}
Real-time Message Handling

Message Controller

@Controller
public class ChatController {
    
    @MessageMapping("/chat.sendMessage")
    @SendTo("/topic/public")
    public ChatMessage sendMessage(@Payload ChatMessage chatMessage) {
        chatMessage.setTimestamp(LocalDateTime.now());
        
        // Save message to database
        ChatMessage savedMessage = chatMessageService.saveMessage(chatMessage);
        
        // Notify online users
        notificationService.notifyOnlineUsers(chatMessage);
        
        return savedMessage;
    }
    
    @MessageMapping("/chat.addUser")
    @SendTo("/topic/public")
    public ChatMessage addUser(@Payload ChatMessage chatMessage, SimpMessageHeaderAccessor headerAccessor) {
        // Add username to web socket session
        headerAccessor.getSessionAttributes().put("username", chatMessage.getSenderName());
        
        // Save user join message
        ChatMessage joinMessage = new ChatMessage();
        joinMessage.setType(MessageType.SYSTEM);
        joinMessage.setContent(chatMessage.getSenderName() + " joined the chat");
        joinMessage.setTimestamp(LocalDateTime.now());
        
        return chatMessageService.saveMessage(joinMessage);
    }
    
    @EventListener
    public void handleWebSocketDisconnectListener(SessionDisconnectEvent event) {
        SimpMessageHeaderAccessor headerAccessor = SimpMessageHeaderAccessor.wrap(event.getMessage());
        String username = (String) headerAccessor.getSessionAttributes().get("username");
        
        if (username != null) {
            ChatMessage chatMessage = new ChatMessage();
            chatMessage.setType(MessageType.SYSTEM);
            chatMessage.setContent(username + " left the chat");
            chatMessage.setTimestamp(LocalDateTime.now());
            
            chatMessageService.saveMessage(chatMessage);
            messagingTemplate.convertAndSend("/topic/public", chatMessage);
        }
    }
}

Real-time Notification Service

@Service
public class NotificationService {
    
    @Autowired
    private SimpMessagingTemplate messagingTemplate;
    
    public void sendNotificationToUser(String userId, Notification notification) {
        // Save notification to database
        notificationRepository.save(notification);
        
        // Send real-time notification
        messagingTemplate.convertAndSendToUser(
            userId,
            "/queue/notifications",
            notification
        );
    }
    
    public void sendNotificationToCourse(String courseId, Notification notification) {
        // Get all enrolled students
        List<Enrollment> enrollments = enrollmentRepository.findByCourseId(courseId);
        
        for (Enrollment enrollment : enrollments) {
            sendNotificationToUser(enrollment.getStudent().getId(), notification);
        }
    }
    
    public void sendSystemNotification(Notification notification) {
        // Get all active users
        List<User> activeUsers = userService.findActiveUsers();
        
        for (User user : activeUsers) {
            sendNotificationToUser(user.getId(), notification);
        }
    }
}

Live Screen Sharing

Screen Sharing Service

@Service
public class ScreenSharingService {
    
    public void startScreenShare(String sessionId, String userId) {
        ScreenShareSession session = new ScreenShareSession();
        session.setSessionId(sessionId);
        session.setHostId(userId);
        session.setStatus(ScreenShareStatus.ACTIVE);
        session.setStartedAt(LocalDateTime.now());
        
        screenShareRepository.save(session);
        
        // Notify participants
        notifyParticipants(sessionId, "Screen sharing started");
    }
    
    public void stopScreenShare(String sessionId) {
        ScreenShareSession session = screenShareRepository.findBySessionId(sessionId);
        if (session != null) {
            session.setStatus(ScreenShareStatus.STOPPED);
            session.setStoppedAt(LocalDateTime.now());
            screenShareRepository.save(session);
            
            // Notify participants
            notifyParticipants(sessionId, "Screen sharing stopped");
        }
    }
    
    private void notifyParticipants(String sessionId, String message) {
        messagingTemplate.convertAndSend("/topic/screenshare/" + sessionId, message);
    }
}

Real-time Progress Updates

Live Progress Tracking

Progress Update Service

@Service
public class LiveProgressService {
    
    public void updateProgress(String courseId, String studentId, double progress) {
        // Update progress in database
        progressTrackingService.updateCourseProgress(courseId, studentId, progress);
        
        // Send real-time update to teacher
        String teacherId = courseService.findById(courseId).getTeacher().getId();
        
        ProgressUpdate update = ProgressUpdate.builder()
            .courseId(courseId)
            .studentId(studentId)
            .progress(progress)
            .timestamp(LocalDateTime.now())
            .build();
        
        messagingTemplate.convertAndSendToUser(
            teacherId,
            "/queue/progress-updates",
            update
        );
    }
    
    public void broadcastCourseProgress(String courseId) {
        List<Enrollment> enrollments = enrollmentRepository.findByCourseId(courseId);
        
        for (Enrollment enrollment : enrollments) {
            double progress = progressTrackingService.getCourseProgress(
                courseId, enrollment.getStudent().getId()).getOverallProgress();
            
            updateProgress(courseId, enrollment.getStudent().getId(), progress);
        }
    }
}

@Component
public class WebSocketEventListener {
    
    @EventListener
    public void handleWebSocketConnectListener(SessionConnectEvent event) {
        SimpMessageHeaderAccessor headerAccessor = SimpMessageHeaderAccessor.wrap(event.getMessage());
        String username = (String) headerAccessor.getSessionAttributes().get("username");
        
        if (username != null) {
            // Update user online status
            userService.updateOnlineStatus(username, true);
            
            // Notify other users
            ChatMessage chatMessage = new ChatMessage();
            chatMessage.setType(MessageType.SYSTEM);
            chatMessage.setContent(username + " is now online");
            chatMessage.setTimestamp(LocalDateTime.now());
            
            messagingTemplate.convertAndSend("/topic/public", chatMessage);
        }
    }
    
    @EventListener
    public void handleWebSocketDisconnectListener(SessionDisconnectEvent event) {
        SimpMessageHeaderAccessor headerAccessor = SimpMessageHeaderAccessor.wrap(event.getMessage());
        String username = (String) headerAccessor.getSessionAttributes().get("username");
        
        if (username != null) {
            // Update user online status
            userService.updateOnlineStatus(username, false);
            
            // Notify other users
            ChatMessage chatMessage = new ChatMessage();

            chatMessage.setContent(username + " is now offline");
            chatMessage.setTimestamp(LocalDateTime.now());

Performance Optimization

WebSocket Performance

Connection Pooling

@Configuration
public class WebSocketPerformanceConfig {
    
    @Bean
    public TaskExecutor messageBrokerTaskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(10);
        executor.setMaxPoolSize(20);
        executor.setQueueCapacity(100);
        executor.setThreadNamePrefix("ws-");
        executor.initialize();
        return executor;
    }
    
    @Bean
    public MessageBrokerRegistry messageBrokerRegistry() {
        MessageBrokerRegistry registry = new MessageBrokerRegistry(null);
        registry.setPreservePublishOrder(true);
        registry.setTaskExecutor(messageBrokerTaskExecutor());
        return registry;
    }
}

Message Rate Limiting

@Component
public class MessageRateLimiter {
    
    private final Map<String, RateLimitInfo> userRateLimits = new ConcurrentHashMap<>();
    private static final int MAX_MESSAGES_PER_MINUTE = 30;
    
    public boolean isRateLimited(String userId) {
        RateLimitInfo info = userRateLimits.get(userId);
        LocalDateTime now = LocalDateTime.now();
        
        if (info == null || info.getLastReset().isBefore(now.minusMinutes(1))) {
            info = new RateLimitInfo(now, 1);
            userRateLimits.put(userId, info);
            return false;
        }
        
        if (info.getMessageCount() >= MAX_MESSAGES_PER_MINUTE) {
            return true;
        }
        
        info.incrementMessageCount();
        return false;
    }
    
    private static class RateLimitInfo {
        private LocalDateTime lastReset;
        private int messageCount;
        
        public RateLimitInfo(LocalDateTime lastReset, int messageCount) {
            this.lastReset = lastReset;
            this.messageCount = messageCount;
        }
        
        public void incrementMessageCount() { messageCount++; }
        public LocalDateTime getLastReset() { return lastReset; }
        public int getMessageCount() { return messageCount; }
    }
}
